### sightly element replacement
#
<div data-sly-element="${'h1'}">Hello, world.</div>
===
<h1>Hello, world.</h1>
#
### ignore forbidden element
#
<div data-sly-element="${'script'}">Hello, world.</div>
===
<div>Hello, world.</div>
#
### empty element
#
<div data-sly-element="${''}">Hello, world.</div>
===
<div>Hello, world.</div>
#
### self closing
#
<foo data-sly-element="${'br'}"/>
===
<br>
